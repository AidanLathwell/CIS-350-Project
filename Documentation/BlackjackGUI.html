 15        width = image.get_width()

 18        self.rect = self.image.get_rect()

 20        self.clicked = False  # for preventing multiple clicks when only clicking once

 22    """ Method will determine if a button gets clicked. """
 23    def draw(self):
 24
 25        action = False
 26
 27        # Get mouse position
 28        pos = pygame.mouse.get_pos()
 29
 30        # Check mouse over button and clicked the button
 31        if self.rect.collidepoint(pos):
 32
 33            # Check if button was clicked (tuple rep. 0 means left click, 1 middle click, 2 right click)
 34            if pygame.mouse.get_pressed()[0] == 1 and self.clicked == False:
 35                self.clicked = True
 36                action = True
 37
 38            # Make it so that button is clicked once instead of multiple times after 1 click
 39            if pygame.mouse.get_pressed()[0] == 0:
 40                self.clicked = False
 41
 42        return action

302    """ Method called after the game ends to determine the winner. """
303    def determine_winner(self):
304
305        if self.game_over and self.number_of_bets_subtracted == 0:
306            self.number_of_bets_subtracted += 1
307            # Check which case applies to the game state and assign results
308            if 21 >= self.player.hand.hand_value > self.dealer.hand.hand_value:
309                self.winner = self.player
310                self.player.balance += self.bet_amount
311                self.dealer.balance -= self.bet_amount
312            elif 21 >= self.dealer.hand.hand_value > self.player.hand.hand_value:
313                self.winner = self.dealer
314                self.player.balance -= self.bet_amount
315                self.dealer.balance += self.bet_amount
316            elif self.player.hand.hand_value > 21 >= self.dealer.hand.hand_value:
317                self.winner = self.dealer
318                self.player.balance -= self.bet_amount
319                self.dealer.balance += self.bet_amount
